闭包

什么是闭包

    function aaa(a){
        
        var b = 5;
    
        function bbb(){
            alert(a);
            alert(b);
        }
    
    }
    
    aaa();

垃圾回收机制

函数嵌套函数,内部函数可以引用外部函数的参数和变量,参数和变量不会被垃圾回收机制所收回

闭包与垃圾回收机制

  • 希望一个变量长期驻扎在内存当中

  • 避免全局变量的污染

    function aaa(){
        var a = 5;
        function bbb(){
            alert(a);
        }
        return bbb;
    }
    
    // 局部变量累加
    var c = aaa();
    
    c();

函数声明和函数表达式

    /*function aaa(){
        alert(1);
    }
    aaa();*/
    
    (function(){
        alert(1);
    })();
  • 私有成员的存在
    var aaa = (function(){
    
        var a = 1;
        
        function bbb(){
            a++;
            alert(a);
        }
        
        function ccc(){
            a++;
            alert(a);
        }
        
        return {
            
            b : bbb,
            c : ccc
            
        }
    
    })();
    
    
    aaa.b();  //2
    aaa.c();  //3
    
    // 外部无法访问到
    //alert(a);
    //alert(bbb);

  • 在循环中直接找到对应元素的索引

    /*for(var i=0;i<aLi.length;i++){
