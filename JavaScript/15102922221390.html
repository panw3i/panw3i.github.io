    对象的组成

    属性就是变量,方法就是函数 ,不过是属于对象的变量和函数

    var arr=[1,2,3,4,5];

var a=12;       //变量：自由
arr.a=5;        //属性：属于一个对象

function show() //函数
{
    alert('a');
}

arr.fn=function ()  //方法
{
    alert('a');
};

    Object对象

    Object 是一个 '空'的对象

    var obj=new Object();

obj.name='blue';
obj.sex='男';

obj.showName=function (){
    alert('我的名字叫：'+this.name);
};
obj.showSex=function (){
    alert('我是'+this.sex+'的');
};

var obj2=new Object();

obj2.name='leo';
obj2.sex='女';

obj2.showName=function (){
    alert('我的名字叫：'+this.name);
};
obj2.showSex=function (){
    alert('我是'+this.sex+'的');
};

obj.showName();
obj.showSex();
obj2.showName();
obj2.showSex();

    构造函数

    用工厂方式构造对象

    • 工厂方式构造对象步骤 : 1.创建空白对象; 2.加工对象; 3.加工出厂对象
    • 工厂方式构造对象缺点 : 1.没有new ; 2.每个对象都保存相同的方法
    //构造函数
function createPerson(name, sex){
    //1.原料
    var obj=new Object();
    
    //2.加工
    obj.name=name;
    obj.sex=sex;
    
    obj.showName=function (){
        alert('我的名字叫：'+this.name);
    };
    obj.showSex=function (){
        alert('我是'+this.sex+'的');
    };
    
    //3.出厂
    return obj;
}

var p1=createPerson('blue', '男');
var p2=createPerson('leo', '女');

alert(p1.showName==p2.showName);

/*p1.showName();
p1.showSex();
p2.showName();
p2.showSex();*/

    继承

    继承的需求

    • 可以给对象添加任意的属性或者方法,但无法给所有类型添加属性
    • 原型的作用类似于类,可以给一类对象添加成员
    • 在JS里面类就是构造函数
    • 构造函数其实就是普通方法,所以它在JS里面的类型是Function
    • 构造函数首字母大写
    //arr1.sum=function ()
Array.prototype.sum=function ()
{
    var result=0;
    var i=0;
    
    for(i=0;i<this.length;i++)
    {
        result+=this[i];
    }
    
    return result;
};

/*alert(arr1.sum());
alert(arr2.sum());*/

    成员方法和成员属性

    相同的东西加到原型里面,不同的东西放到构造函数里

    function Person(name, sex)
{
    this.name=name;
    this.sex=sex;
}

Person.prototype.showName=function ()
{
    alert(this.name);
};

Person.prototype.showSex=function ()
{
    alert(this.sex);
};

var p=new Person('blue', '男');

p.showName();
p.showSex();

    成员的优先级

    对象成员会覆盖原型的成员,但原型的成员并不会消失

    Array.prototype.a=12;  // 此处的a相当于类属性

var arr=[1,2,3];

alert(arr.a);   //12

arr.a=5;

alert(arr.a);   //5

delete arr.a;

alert(arr.a);   //12
