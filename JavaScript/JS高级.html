闭包

什么是闭包

    function aaa(a){
        
        var b = 5;
    
        function bbb(){
            alert(a);
            alert(b);
        }
    
    }
    
    aaa();

垃圾回收机制

函数嵌套函数,内部函数可以引用外部函数的参数和变量,参数和变量不会被垃圾回收机制所收回

闭包与垃圾回收机制

  • 希望一个变量长期驻扎在内存当中

  • 避免全局变量的污染

    function aaa(){
        var a = 5;
        function bbb(){
            alert(a);
        }
        return bbb;
    }
    
    // 局部变量累加
    var c = aaa();
    
    c();

函数声明和函数表达式

    /*function aaa(){
        alert(1);
    }
    aaa();*/
    
    (function(){
        alert(1);
    })();
  • 私有成员的存在
    var aaa = (function(){
    
        var a = 1;
        
        function bbb(){
            a++;
            alert(a);
        }
        
        function ccc(){
            a++;
            alert(a);
        }
        
        return {
            
            b : bbb,
            c : ccc
            
        }
    
    })();
    
    
    aaa.b();  //2
    aaa.c();  //3
    
    // 外部无法访问到
    //alert(a);
    //alert(bbb);

  • 模块化代码

  • 在循环中直接找到对应元素的索引

    var aLi = document.getElementsByTagName('li');
    
    
    /* 按照传统的for循环注册点击事件,点击事件并没有立即执行,当点击事件发生时,
    函数查找变量i的值,i的值为最后一次i的赋值 */
    /*for(var i=0;i<aLi.length;i++){
    
        (function(i){
        
            aLi[i].onclick = function(){
                alert(i);
            };
        
        })(i);
        
        
    }*/
    
    for(var i=0;i<aLi.length;i++){
    
    // 参数形成闭包
    
            aLi[i].onclick = (function(i){   //2 
            
                return function(){
                    alert(i);    // 3
                }
                
            })(i);  // 1 

        
    }
    
};

// 2处是函数 声明 时的要传入的参数 , 1 是接受传入的参数 , 3 同部函数要访问的变量i
