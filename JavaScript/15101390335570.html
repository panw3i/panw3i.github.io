window.onload=function (){

    var oUl=document.getElementById('ul1');   // document是顶级节点  var aLi=oUl.getElementsByTagName('li');
    var i=0;
    
    for(i=0;i<aLi.length;i++){
    
        if(aLi[i].className=='box'){
        
            aLi[i].style.background='red';
        }
    }

window.onload=function ()
{
    var oUl=document.getElementById('ul1');

获得子节点的父节点


window.onload=function ()
{
    var aA=document.getElementsByTagName('a');
    var i=0;
    
    for(i=0;i<aA.length;i++)
    {
        aA[i].onclick=function ()
        {
            this.parentNode.style.display='none';
        }
    }
}

DOM操作操作节点的属性

三个方法能用

window.onload=function ()
{
    var oTxt=document.getElementById('txt1');
    
    // 方法一
    oTxt.value='123';
    
    // 方法二
    oTxt['value']='abc';      // 属性值一定是字符串
    
    // 方法三
    oTxt.setAttribute('value', 'rtertw');
    
    alert(oTxt.getAttribute('id'));  // alert(oTxt.removeAttribute('id'));
}

通过父节点查找指定子节点的函数封装

function getByClass(oParent, sClass)
{
    var aEle=oParent.getElementsByTagName('*');
    var aResult=[];
    var i=0;
    
    for(i=0;i<aEle.length;i++)
    {
        if(aEle[i].className==sClass)
        {
            aResult.push(aEle[i]);
        }
    }
    
    return aResult;
}

window.onload=function ()
{
    var oUl=document.getElementById('ul1');
    var aBox=getByClass(oUl, 'box');
    var i=0;
    
    for(i=0;i<aBox.length;i++)
    {
        aBox[i].style.background='yellow';
    }
}

创建节点并添加到父节点

window.onload=function (){
    var oBtn=document.getElementById('btn1');
    var oUl=document.getElementById('ul1');
    
    oBtn.onclick=function (){
        var oLi=document.createElement('li');
        
        //父.appendChild(子节点)
        oUl.appendChild(oLi);
    }
}

为子节点增加内容节点

window.onload=function (){
    var oBtn=document.getElementById('btn1');
    var oTxt=document.getElementById('txt1');
    var oUl=document.getElementById('ul1');
    
    oBtn.onclick=function (){
        var oLi=document.createElement('li');
        
        oLi.innerHTML=oTxt.value;
        
        //父.appendChild(子节点)
        oUl.appendChild(oLi);
    }
}

在子节点前/后增加子节点

window.onload=function (){
    var oBtn=document.getElementById('btn1');
    var oTxt=document.getElementById('txt1');
    var oUl=document.getElementById('ul1');
    
    oBtn.onclick=function (){
        var oLi=document.createElement('li');
        var aLi=oUl.getElementsByTagName('li');
        
        oLi.innerHTML=oTxt.value;
        
        // 父.insertBefore(新增加的子节点,指定已存在的子节点 
        oUl.insertBefore(oLi, aLi[0]);
    }
}

判断子节点是否存在

window.onload=function (){
    var oBtn=document.getElementById('btn1');
    var oTxt=document.getElementById('txt1');
    var oUl=document.getElementById('ul1');
    
    oBtn.onclick=function (){
        var oLi=document.createElement('li');
        var aLi=oUl.getElementsByTagName('li');
        
        oLi.innerHTML=oTxt.value;
        
        // 判断是父节点是否存在子节点,如果没有就直接添加,否则找到指定子节点进行插入操作
        if(aLi.length==0){
            oUl.appendChild(oLi);
        }
        else{
            oUl.insertBefore(oLi, aLi[0]);
        }
    }
}

删除节点

window.onload=function ()
{
    var aA=document.getElementsByTagName('a');
    var oUl=document.getElementById('ul1');
    var i=0;
    
    for(i=0;i<aA.length;i++)
    {
        aA[i].onclick=function ()
        {
        
        // 父节点.removeChild(子节点)

            oUl.removeChild(this.parentNode);   // 查找子节点的直接父节点
        }
    }
}
